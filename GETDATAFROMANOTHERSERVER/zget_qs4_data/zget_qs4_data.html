Description: GET DATA FROM QUALITY SERVER 
FUNCTION ZGET_QS4_DATA.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     VALUE(IM_TABLE) TYPE  STRING
*"     VALUE(IM_WHERE) TYPE  STRING
*"  EXPORTING
*"     VALUE(ET_JSON) TYPE  STRING
*"----------------------------------------------------------------------

*       Global data declarations

  DATA: lr_data TYPE REF TO data.
  FIELD-SYMBOLS: <lt_data> TYPE ANY TABLE.

  " 1. Dynamically create the table structure
  TRY.
      CREATE DATA lr_data TYPE TABLE OF (im_table).
      ASSIGN lr_data->* TO <lt_data>.
    CATCH cx_sy_create_data_error.
      et_json = '{ "error": "Invalid table name" }'.
      RETURN.
  ENDTRY.

  " 2. Select data (Limiting to 100 rows for performance)
  SELECT * FROM (im_table)
    WHERE (im_where)
    INTO TABLE @<lt_data>.

  " 3. Check if data was found
  IF sy-subrc = 0 AND <lt_data> IS ASSIGNED.
    " 4. Serialize to JSON
    /ui2/cl_json=>serialize(
      EXPORTING
        data          = <lt_data>
        compress      = abap_true  " Removes empty fields to save space
        pretty_name   = /ui2/cl_json=>pretty_mode-camel_case
      RECEIVING
        r_json        = et_json ).
  ELSE.
    et_json = '{ "info": "No data found in table" }'.
  ENDIF.

ENDFUNCTION.
