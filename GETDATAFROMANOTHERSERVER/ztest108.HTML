Description: tEST FM FOR GETTING DATA FROM QS4 
*&---------------------------------------------------------------------*
*& Report ZTEST108
*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
REPORT ztest108.

PARAMETERS: p_table TYPE string DEFAULT 'T001',
            p_dest  TYPE rfcdest DEFAULT 'QS4100CLNTEST',
            p_key   TYPE string DEFAULT 'MATNR',
            p_value TYPE string DEFAULT 'INTERNAL FORMAT'.

DATA: lv_json_result TYPE string,
      lv_msg         TYPE c LENGTH 255.
DATA: lr_data TYPE REF TO data.
FIELD-SYMBOLS: <lt_data> TYPE ANY TABLE.

START-OF-SELECTION.

  DATA(lv_where) = |{ p_key } = '{ p_value }'|.
  IF lv_where CP sy-abcde.
    CLEAR lv_where.
  ENDIF.
  TRY.
      CALL FUNCTION 'ZGET_QS4_DATA'
        DESTINATION p_dest
        EXPORTING
          im_table              = p_table
          im_where              = lv_where
        IMPORTING
          et_json               = lv_json_result
        EXCEPTIONS
          system_failure        = 1 MESSAGE lv_msg
          communication_failure = 2 MESSAGE lv_msg
          OTHERS                = 3.


      IF sy-subrc = 0 AND lv_json_result IS NOT INITIAL.

        TRY.
            CREATE DATA lr_data TYPE TABLE OF (p_table).
            ASSIGN lr_data->* TO <lt_data>.
          CATCH cx_sy_create_data_error.
            WRITE: / 'Error: Table structure not found in this system.'.
            RETURN.
        ENDTRY.

        /ui2/cl_json=>deserialize(
          EXPORTING
            json         = lv_json_result
            pretty_name  = /ui2/cl_json=>pretty_mode-camel_case
          CHANGING
            data         = <lt_data> ).

        IF <lt_data> IS ASSIGNED AND lines( <lt_data> ) > 0.
          " Quick ALV Grid display to see the actual table data
          cl_demo_output=>display( <lt_data> ).
        ELSE.
          WRITE: / 'Deserialization failed or JSON was empty.'.
        ENDIF.

      ELSE.
        WRITE: / 'RFC Error or No Data returned.'.
      ENDIF.
    CATCH cx_root INTO DATA(lo_data).
      cl_demo_output=>display( lo_data ).
  ENDTRY.
